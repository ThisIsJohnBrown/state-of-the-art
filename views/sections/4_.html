Three.js 


var geometry = new THREE.BoxGeometry( 1, 1, 1 );
var material = new THREE.MeshBasicMaterial( { color: 0x00ff00 } );
var cube = new THREE.Mesh( geometry, material );
scene.add( cube ); 

function render() {
    cube.rotation.x += 0.1;
    cube.rotation.y += 0.1;
    requestAnimationFrame( render );
    renderer.render( scene, camera );
}
